server {
    server_name management.phungnguyengarment.vn www.management.phungnguyengarment.vn;

    location / {
        proxy_pass http://localhost:8000; # Port ứng dụng Node.js đang chạy
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
    }

    listen 443 ssl; # managed by Certbot
    ssl_certificate /etc/letsencrypt/live/phungnguyengarment.vn/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/phungnguyengarment.vn/privkey.pem; # managed by Certbot
    include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot


}
server {
    if ($host = www.management.phungnguyengarment.vn) {
        return 301 https://$host$request_uri;
    } # managed by Certbot


    if ($host = management.phungnguyengarment.vn) {
        return 301 https://$host$request_uri;
    } # managed by Certbot


    listen 80;
    server_name management.phungnguyengarment.vn www.management.phungnguyengarment.vn;
    return 404; # managed by Certbot




}
